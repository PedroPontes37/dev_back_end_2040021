CREATE database projeto_backend;

CREATE table videos(
	id int auto_increment,
	uploader varchar(45),
	title varchar(45),
	descriptionn varchar(45),
	duration int,
	url varchar(45),
	views int,
	likes int,
	dislikes int,
	comments varchar(45),
	tags varchar(45),
    	PRIMARY KEY (id)
);
